4. ZERO-COST IMAGE MAPPING SYSTEM

Create a file:

Structure:

const imageMap = {
"screen_time": [
"/images/screen1.png",
"/images/screen2.png"
],
"tantrum": [
"/images/tantrum1.png",
"/images/tantrum2.png"
],
"sleep": [
"/images/sleep1.png"
],
"default": [
"/images/default.png"
]
};

function getImages(problem) {
const key = problem.toLowerCase().replace(/\s+/g, "_");

return imageMap[key] || imageMap["default"];
}

module.exports = { getImages };
